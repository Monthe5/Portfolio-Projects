import pandas as pd
from sklearn.linear_model import LinearRegression
import matplotlib.pyplot as plt

# Assuming you have loaded and preprocessed the data into the 'df' DataFrame.
# Make sure the 'Period' column is of datetime type.

# Prepare the data
df['ds'] = pd.to_datetime(df['period'])
df_model = df[['ds', 'Acute Malnutrition']].copy()  # Use .copy() to avoid the SettingWithCopyWarning

# Extract year and month features for the linear regression model using .loc
df_model.loc[:, 'year'] = df_model['ds'].dt.year
df_model.loc[:, 'month'] = df_model['ds'].dt.month

# Train-test split
train = df_model[df_model['ds'] < '2023-01-01']
test = df_model[df_model['ds'] >= '2023-01-01']

# Check if there are samples in the training data
if not train.empty:
    # Create and fit the linear regression model
    model = LinearRegression()
    model.fit(train[['year', 'month']], train['Acute Malnutrition'])

    # Make predictions for the next year (12 months)
    next_year_dates = pd.date_range(start='2023-01-01', periods=12, freq='M')
    next_year_df = pd.DataFrame({'ds': next_year_dates})
    next_year_df.loc[:, 'year'] = next_year_df['ds'].dt.year
    next_year_df.loc[:, 'month'] = next_year_df['ds'].dt.month

    predictions = model.predict(next_year_df[['year', 'month']])

    # Plot the predictions
    plt.figure(figsize=(10, 6))
    plt.plot(df_model['ds'], df_model['Acute Malnutrition'], label='Actual')
    plt.plot(next_year_df['ds'], predictions, label='Predicted', linestyle='dashed')
    plt.xlabel('Date')
    plt.ylabel('Number of Children with Acute Malnutrition')
    plt.title('Prediction for Children with Acute Malnutrition')
    plt.legend()
    plt.show()
else:
    print("Training data is empty. Check the data filtering condition.")
